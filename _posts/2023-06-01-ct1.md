## 내 풀이
* 이 문제는 각 번호키의 좌표를 먼저 저장한다.
* 이후 왼손과 오른손의 좌표를 업데이트하며 문제를 따라가면 쉽게 풀 수 있다.


```python
def solution(numbers, hand):
    
    codi_dic = {1:[0,0], 2:[0,1], 3:[0,2], 
                4:[1,0], 5:[1,1], 6:[1,2],
                7:[2,0], 8:[2,1], 9:[2,2],
                0:[3,1]}
    answer = []
    left_loc = [3, 0]
    right_loc = [3, 2]

    for n in numbers:
        if n in [1, 4, 7]:
            answer.append('L')
            left_loc = codi_dic[n]
        elif n in [3, 6, 9]:
            answer.append('R')
            right_loc = codi_dic[n]
        elif n in [2, 5, 8, 0]:
            left_len = abs(left_loc[0]-codi_dic[n][0]) + abs(left_loc[1]-codi_dic[n][1])
            right_len = abs(right_loc[0]-codi_dic[n][0]) + abs(right_loc[1]-codi_dic[n][1])
            if left_len > right_len:
                answer.append('R')
                right_loc = codi_dic[n]

            elif left_len < right_len:
                answer.append('L')
                left_loc = codi_dic[n]

            elif left_len == right_len:
                if hand=='left':
                    answer.append('L')
                    left_loc = codi_dic[n]
                else:
                    answer.append('R')
                    right_loc = codi_dic[n]

    return ''.join(answer)
```
