## 내 풀이
* 이 문제는 문제를 따라가면서 풀면 쉽게 풀 수 있다.

```python
def convert_binary(num):
    binary = []
    while True:
        remainder = num % 2
        num = num // 2
        binary.append(remainder)
        
        if num < 2:
            binary.append(num)
            break
    
    binary.reverse()
    return int(''.join(map(str, binary)))

def solution(s):
    answer = []
    count = 0
    bin_count = 0
    while True:
        count += s.count('0')
        if s.count('0')!=0:
            s = s.replace('0', '')

        s = len(s)
        s = convert_binary(s)

        s = str(s)
        bin_count += 1

        if s == '1':
            break
            
    answer.append(bin_count)
    answer.append(count)
    
    return answer
```
